Login as ec2-user into Redhat Linux SErver
sudo su -
cd /opt
In /opt folder create maven folder and download maven into this(https://maven.apache.org/download.cgi)
mkdir maven
pwd
   /opt/maven
cd maven
  yum install wget -y
  yum install https://downloads.apache.org/maven/maven-3/3.6.3/binaries/apache-maven-3.6.3-bin.tar.gz

ls
    apache-maven-3.6.3-bin.tar.gz

Untar this file
    tar -xvzf apache-maven-3.6.3-bin.tar.gz

ls
    apache-maven-3.6.3-bin.tar
    apache-maven-3.6.3-bin.tar.gz
